CREATE TABLE IF NOT EXISTS usuarios (
    id SERIAL PRIMARY KEY,
    nome_completo VARCHAR(255) NOT NULL,
    idade VARCHAR(10),
    email VARCHAR(255) UNIQUE NOT NULL,
    estado VARCHAR(50),
    cidade VARCHAR(50),
    endereco TEXT,
    telefone VARCHAR(20) UNIQUE,
    nome_usuario VARCHAR(50) UNIQUE NOT NULL,
    senha VARCHAR(255)
);
